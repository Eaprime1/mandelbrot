```python
class PaymentPlan:
    def calculate_installments(self, total_cost, months):
        if total_cost < 0:
            raise ValueError("total_cost must be non-negative")
        if months <= 0:
            raise ValueError("months must be a positive integer")
        interest_rate = 0.05
        total_with_interest = total_cost * (1 + interest_rate)
        monthly_payment = total_with_interest / months
        return {
            'monthly_amount': round(monthly_payment, 2),
            'total_cost': round(total_with_interest, 2),
            'duration': months
        }
```
